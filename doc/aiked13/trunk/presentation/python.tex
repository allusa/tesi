class RRD(object):
    def __init__(self,delta,f):
        self.B = Buffer(delta,f)
        self.A = []

    def afegeix_disc(self,delta,k,f):
        self.A.append( DiscRoundRobin(delta,k,f) )

    def afegeix(self,m):
        self.B.afegeix(m)

    def consolidable(self):
        return self.B.consolidable()

    def consolidar(self):
        m = self.B.consolida()
        for R in self.A:
            R.afegeix(m)


\end{lstlisting}
